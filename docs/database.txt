  code text NOT NULL,
  name text,
  pin text,
  seat numeric,
  passport_image_url text,
  passport_id text,
  passport_exp_date date,
  visa_image_url text,
  tel text,
  tshirt_size text,
  is_sub_row boolean DEFAULT false,
  parent_code text,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone,
  nationality text,
  CONSTRAINT tour_seat_check_pkey PRIMARY KEY (id),
  CONSTRAINT tour_seat_check_parent_code_fkey FOREIGN KEY (parent_code) REFERENCES public.tour_seat_check(code)
);
